        <Line 
            X1="0"
            Y1="0"
            X2="100"
            Y2="50"
             Stroke="Black"
            StrokeThickness="10"/>
#################################################


Meaning
Stroke =color of line
StrokeThickness = line thickness

            X1="0"
            Y1="0"
            X2="100"
